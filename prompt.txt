-- =========================================
-- CARD
-- =========================================
CREATE TABLE card (
                      id SERIAL PRIMARY KEY,
                      uuid UUID NOT NULL UNIQUE,
                      total_points INTEGER,
                      available_points INTEGER,
                      used_points INTEGER
);

-- =========================================
-- BAG
-- =========================================
CREATE TABLE bag (
                     id SERIAL PRIMARY KEY,
                     uuid UUID NOT NULL UNIQUE
);


-- =========================================
-- DIARY
-- =========================================
CREATE TABLE diary (
                       id SERIAL PRIMARY KEY,
                       uuid UUID NOT NULL DEFAULT gen_random_uuid(),
                       name VARCHAR(100) NOT NULL UNIQUE
);

-- =========================================
-- DIARY_ENTRY
-- =========================================
CREATE TABLE diary_entry (
                             id SERIAL PRIMARY KEY,
                             diary_id INTEGER NOT NULL REFERENCES diary(id) ON DELETE CASCADE,
                             date TIMESTAMP NOT NULL DEFAULT NOW(),
                             description TEXT
);

-- =========================================
-- DIARY_ENTRY_PARTICIPANTS (ManyToMany Player â†” DiaryEntry)
-- =========================================
CREATE TABLE diary_entry_participants (
                                          entry_id INTEGER NOT NULL REFERENCES diary_entry(id) ON DELETE CASCADE,
                                          player_id INTEGER NOT NULL REFERENCES player(id) ON DELETE CASCADE,
                                          PRIMARY KEY (entry_id, player_id)
);

-- =========================================
-- DIARY_IMAGE
-- =========================================
CREATE TABLE diary_image (
                             id SERIAL PRIMARY KEY,
                             entry_id INTEGER NOT NULL REFERENCES diary_entry(id) ON DELETE CASCADE,
                             image_data BYTEA NOT NULL
);

-- =========================================
-- PLAYER
-- =========================================
CREATE TABLE player (
                        id SERIAL PRIMARY KEY,
                        uuid UUID NOT NULL UNIQUE,
                        pg_name VARCHAR(50),
                        background TEXT,
                        diary_id INTEGER,
                        user_id BIGINT NOT NULL,
                        card_id INTEGER,
                        bag_id INTEGER NOT NULL UNIQUE,
                        CONSTRAINT fk_player_diary FOREIGN KEY (diary_id) REFERENCES diary (id) ON DELETE CASCADE,
                        CONSTRAINT fk_player_user FOREIGN KEY (user_id) REFERENCES users (id) ON DELETE CASCADE,
                        CONSTRAINT fk_player_card FOREIGN KEY (card_id) REFERENCES card (id),
                        CONSTRAINT fk_player_bag FOREIGN KEY (bag_id) REFERENCES bag (id)
);

-- =========================================
-- MONEY
-- =========================================
CREATE TABLE money (
                       id SERIAL PRIMARY KEY,
                       uuid UUID NOT NULL UNIQUE,
                       money_name VARCHAR(50),
                       bag_id INTEGER NOT NULL,
                       CONSTRAINT fk_money_bag FOREIGN KEY (bag_id) REFERENCES bag (id) ON DELETE CASCADE
);

-- =========================================
-- REAGENT
-- =========================================
CREATE TABLE reagent (
                         id SERIAL PRIMARY KEY,
                         uuid UUID NOT NULL UNIQUE,
                         reagent_name VARCHAR(50),
                         bag_id INTEGER NOT NULL,
                         CONSTRAINT fk_reagent_bag FOREIGN KEY (bag_id) REFERENCES bag (id) ON DELETE CASCADE
);

-- =========================================
-- KINGDOM
-- =========================================
CREATE TABLE kingdom (
                         id SERIAL PRIMARY KEY,
                         uuid UUID NOT NULL UNIQUE,
                         kingdom_name VARCHAR(50),
                         player_id INTEGER,
                         CONSTRAINT fk_kingdom_player FOREIGN KEY (player_id) REFERENCES player (id) ON DELETE CASCADE
);

-- =========================================
-- ABILITY
-- =========================================
CREATE TABLE ability (
                         id SERIAL PRIMARY KEY,
                         uuid UUID NOT NULL UNIQUE,
                         ability_name VARCHAR(50),
                         background TEXT
);

-- =========================================
-- PLAYER_ABILITY (join table ManyToMany)
-- =========================================
CREATE TABLE player_ability (
                                player_id INTEGER NOT NULL,
                                ability_id INTEGER NOT NULL,
                                PRIMARY KEY (player_id, ability_id),
                                CONSTRAINT fk_player_ability_player FOREIGN KEY (player_id) REFERENCES player (id) ON DELETE CASCADE,
                                CONSTRAINT fk_player_ability_ability FOREIGN KEY (ability_id) REFERENCES ability (id) ON DELETE CASCADE
);
